``` r
fit_brms_reg_s6 <-
  brm(yi | se(sei) ~
        1 + YEAR + TEST + OPT_AREA + COUNTRY + 
        (1 | ID),
      chains = 5, iter = 5000, warmup = 3000,
      prior = prior(normal(0,1), class = sd),
      control = list(adapt_delta=0.85, max_treedepth = 20),
      cores = 5,
      data = subset(es, as.integer(FLAG) == 1),
      open_progress = FALSE,
      seed = 7)
```
