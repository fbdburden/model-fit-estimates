  mutate(yi = case_when (VALUE_X == 0 ~ -15, TRUE ~ yi ))

# BRMS: Version 9

``` r
## fit model
fit_brms_reg_s9 <-
  brm(yi | se(sei) ~
        1 + YEAR + TEST + OPT_AREA +
          (1 | COUNTRY) +
          (1 | COUNTRY:ID)+
          (1 | COUNTRY:ID:DTP_ID),
      chains = 5, iter = 5000, warmup = 3000,
      prior = prior(normal(0,1), class = sd),
      control = list(adapt_delta=0.95, max_treedepth = 20),
      cores = 5,
      data = subset(es, as.integer(FLAG) == 1),
      open_progress = FALSE,
      seed = 7)
```
