``` r
fit_brms_reg_s11 <-
  brm(yi | se(sei) ~
       1 + YEAR + 
        (1  | REG2) +
        (1  | REG2:SUB2) +
        (1  | REG2:SUB2:COUNTRY) +
        (1  | REG2:SUB2:COUNTRY:ID) +
        (1  | REG2:SUB2:COUNTRY:ID:DTP_ID),
      chains = 5, iter = 5000, warmup = 3000,
      prior = prior(normal(0,1), class = sd),
      cores = 5,
      data = subset(es, as.integer(FLAG) == 1), 
      open_progress = FALSE,
      control = list(adapt_delta=0.99, max_treedepth = 20),
      seed =7 )
```
