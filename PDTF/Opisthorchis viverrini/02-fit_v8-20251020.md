# BRMS

``` r
fit_brms_reg_s8  <-
  brm(yi | se(sei) ~
         1 + YEAR + TEST + REF_LOC_LEVEL +
        (1 | COUNTRY) + 
        (1 | COUNTRY:ID ) ,
      chains = 5, iter = 5000, warmup = 3000,
     prior = prior(normal(0,1), class = sd),
      control = list(adapt_delta=0.99, max_treedepth = 20),
      cores = 5,
      data = subset(es, as.integer(FLAG) == 1),
      open_progress = FALSE,
      seed = 7)
```
