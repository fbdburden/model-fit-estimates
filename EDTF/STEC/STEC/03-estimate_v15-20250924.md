``` r
## set up dataframe
sim_all <-
  data.frame(
    sei = 0,
    SEROTYPE_O157 = rep(0:1, each = 2),
    VALUE_CASES_AGE2 = c(1,3, 1,3))

                                   levels = c(1,3), 
                                   labels = c("Age 0-19", "Age 20-125"))

                                levels = c(0,1),
                                labels = c("non-O157", "O157"))
```
