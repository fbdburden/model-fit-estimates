fit_brms_reg_s15 <-
  brm(yi | se(sei) ~
        1 + SEROTYPE_O157 +
        VALUE_CASES_AGE2 +
        (1 | ID) + 
        (1 | ID:DTP_ID),
      chains = 5, iter = 7000, warmup = 3000,
      cores = 5,
      prior = prior(normal(0,1), class = sd),
      data = es,
      open_progress = FALSE,
      seed = 7)
