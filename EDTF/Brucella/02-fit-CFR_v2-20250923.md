## BRMS model: Version 2

``` r
fit_brms_reg_CFR_s2 <-
  brm(yi | se(sei) ~
        1 +
        (1  | ID) +
        (1  | ID:DTP_ID),
      chains = 5, iter = 5000, warmup = 3000,
      cores = 5,
      prior = prior(normal(0,1), class = sd),
      data = es,
      open_progress = FALSE,
      # control=list(adapt_delta = 0.95, max_treedepth = 20),
      seed =7 )
```
